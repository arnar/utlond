module Sos where
import Ast
import State
import Eval

import Random
import Control.Monad

pick :: [a] -> IO a
pick l = do r <- getStdRandom (randomR (1, length l))
            return $ l !! (r-1)

data Action
    = Epsilon
    | Output String Integer
    | Input String String

data Config
    = Inter C State Action
    | Final State Action
    | FailedGC

--isFinal :: Config -> Bool
--isFinal (Inter c s) = False
--isFinal (Final s) = True

mentions :: Action -> String -> Bool
mentions Epsilon c = False
mentions (Output s i) c = s == c
mentions (Input s v) c = s == c

derivSeq :: Config -> IO [Config]
derivSeq (Inter c s a) = (Inter c s a :) `liftM` (derivSeq =<< sosC (Inter c s a))
derivSeq (Final   s a) = return [Final s a]


sosC :: Config -> IO Config

sosC (Inter Nop s a) = return (Final s Epsilon)

sosC (Inter (Assign x e) s a) =
    return (Final (update s x (evalA e s)) Epsilon)

sosC (Inter (Compose c1 c2) s a) =
    sosC (Inter c1 s a) >>= return . f
    where
        f (Final s' a') = Inter c2 s' a'
        f (Inter c1' s' a') = Inter (Compose c1' c2) s' a'

sosC (Inter (If gc) s a) =
    sosGC gc s a >>= return . f
    where
        f (Inter c s' a') = Inter c s' a'

sosC (Inter (Do gc) s a) =
    sosGC gc s a >>= return . f
    where
        f (Inter c s' a') = Inter (Compose c (Do gc)) s a'
        f FailedGC = Final s Epsilon

sosC (Inter (Star c) s a)
    = pick [
        Inter (Compose c (Star c)) s a,
        Final s a
    ]

sosC (Inter (ChanCommand (Send c e)) s a) =
    return $ Final s (Output c (evalA e s))

sosC (Inter (ChanCommand (Receive c var)) s a) =
    return $ Final s (Input c var)

sosC (Inter (Par c1 c2) s a) = do
    cfg1 <- sosC $ Inter c1 s a
    cfg2 <- sosC $ Inter c2 s a
    f cfg1 cfg2
    where
        f (Final s1 (Output ch1 val)) (Final s2 (Input ch2 var))
            | ch1 == ch2 = return $ Final (update s2 var val) Epsilon
            | otherwise = g
        f (Final s1 (Input ch1 var)) (Final s2 (Output ch2 val))
            | ch1 == ch2 = return $ Final (update s1 var val) Epsilon
            | otherwise = g
        f (Inter c s1 (Output ch1 val)) (Final s2 (Input ch2 var))
            | ch1 == ch2 = return $ Inter c (update s2 var val) Epsilon
            | otherwise = g
        f (Inter c s1 (Input ch1 var)) (Final s2 (Output ch2 val))
            | ch1 == ch2 = return $ Inter c (update s1 var val) Epsilon
            | otherwise = g
        f (Final s1 (Output ch1 val)) (Inter c s2 (Input ch2 var))
            | ch1 == ch2 = return $ Inter c (update s2 var val) Epsilon
            | otherwise = g
        f (Final s1 (Input ch1 var)) (Inter c s2 (Output ch2 val))
            | ch1 == ch2 = return $ Inter c (update s1 var val) Epsilon
            | otherwise = g
        f (Inter c1' s1 (Output ch1 val)) (Inter c2' s2 (Input ch2 var))
            | ch1 == ch2 = return $ Inter (Par c1' c2') (update s2 var val) Epsilon
            | otherwise = g
        f (Inter c1' s1 (Input ch1 var)) (Inter c2' s2 (Output ch2 val))
            | ch1 == ch2 = return $ Inter (Par c1' c2') (update s1 var val) Epsilon
            | otherwise = g
        f _ _ = g
        g = do
            (i,selectedC) <- pick [(1,c1), (2,c2)]
            cfg <- sosC $ Inter selectedC s a
            case (i,cfg) of
                (1, Final   s a) -> return $ Inter c2 s a
                (1, Inter c s a) -> return $ Inter (Par c c2) s a
                (2, Final   s a) -> return $ Inter c1 s a
                (2, Inter c s a) -> return $ Inter (Par c1 c) s a

sosC (Inter (Local ch c) s a) =
    f =<< (sosC $ Inter c s a)
    where
        f (Final s a)
            | not (a `mentions` ch) = return $ Final s a
        f (Inter c s a)
            | not (a `mentions` ch) = return $ Inter (Local ch c) s a

sosGC :: GC -> State -> Action -> IO Config

sosGC (GuardedCommand (BoolGuard b) c) s a
    | evalB b s = return $ Inter c s a
    | otherwise = return FailedGC

sosGC (GuardedCommand (ChanGuard (Send ch e)) c) s a =
    return $ Inter c s (Output ch (evalA e s))

sosGC (GuardedCommand (ChanGuard (Receive ch var)) c) s a =
    return $ Inter c s (Input ch var)

sosGC (GuardedCommand (CompositeGuard b chop) c) s a
    | evalB b s = sosGC (GuardedCommand (ChanGuard chop) c) s a
    | otherwise = return FailedGC

sosGC (SelectGC g1 g2) s a =
    pickOrFail . filter (not . fails) =<< mapM (\g -> sosGC g s a) gcList
    where
        gcList = flattenGC (SelectGC g1 g2)
        flattenGC (SelectGC g1 g2) = (flattenGC g1) ++ (flattenGC g2)
        flattenGC a = [a]
        fails FailedGC = True
        fails _ = False
        pickOrFail [] = return FailedGC
        pickOrFail l = pick l

As well as creating your own lexer, writing a new formatter for Pygments is easy and straightforward.

A formatter is a class that is initialized with some keyword arguments (the formatter options) and that must provides a format() method. Additionally a formatter should provide a get_style_defs() method that returns the style definitions from the style in a form usable for the formatter's output format.

Quickstart

The most basic formatter shipped with Pygments is the NullFormatter. It just sends the value of a token to the output stream:

from pygments.formatter import Formatter

class NullFormatter(Formatter):
    def format(self, tokensource, outfile):

        for ttype, value in tokensource:
            outfile.write(value)

As you can see, the format() method is passed two parameters: tokensource and outfile. The first is an iterable of (token_type, value) tuples, the latter a file like object with a write() method.

Because the formatter is that basic it doesn't overwrite the get_style_defs() method.

Styles

Styles aren't instantiated but their metaclass provides some class functions so that you can access the style definitions easily.

Styles are iterable and yield tuples in the form (ttype, d) where ttype is a token and d is a dict with the following keys:

'color'
Hexadecimal color value (eg: 'ff0000' for red) or None if not defined.
'bold'
True if the value should be bold
'italic'
True if the value should be italic
'underline'
True if the value should be underlined
'bgcolor'
Hexadecimal color value for the background (eg: 'eeeeeee' for light gray) or None if not defined.
'border'
Hexadecimal color value for the border (eg: '0000aa' for a dark blue) or None for no border.

Additional keys might appear in the future, formatters should ignore all keys they don't support.

HTML 3.2 Formatter

For an more complex example, let's implement a HTML 3.2 Formatter. We don't use CSS but inline markup (<u>, <font>, etc). Because this isn't good style this formatter isn't in the standard library ;-)

from pygments.formatter import Formatter

class OldHtmlFormatter(Formatter):

    def __init__(self, **options):

        Formatter.__init__(self, **options)

        # create a dict of (start, end) tuples that wrap the
        # value of a token so that we can use it in the format

        # method later
        self.styles = {}

        # we iterate over the `_styles` attribute of a style item
        # that contains the parsed style values.

        for token, style in self.style:
            start = end = ''

            # a style item is a tuple in the following form:
            # colors are readily specified in hex: 'RRGGBB'
            if style['color']:
                start += '<font color="#%s">' % style['color']

                end = '</font>' + end
            if style['bold']:

                start += '<b>'
                end = '</b>' + end

            if style['italic']:
                start += '<i>'
                end = '</i>' + end

            if style['underline']:
                start += '<u>'
                end = '</u>' + end

            self.styles[token] = (start, end)

    def format(self, tokensource, outfile):
        # lastval is a string we use for caching

        # because it's possible that an lexer yields a number
        # of consecutive tokens with the same token type.
        # to minimize the size of the generated html markup we
        # try to join the values of same-type tokens here
        lastval = ''

        lasttype = None

        # wrap the whole output with <pre>
        outfile.write('<pre>')

        for ttype, value in tokensource:
            # if the token type doesn't exist in the stylemap

            # we try it with the parent of the token type
            # eg: parent of Token.Literal.String.Double is
            # Token.Literal.String
            while ttype not in self.styles:

                ttype = ttype.parent
            if ttype == lasttype:

                # the current token type is the same of the last
                # iteration. cache it
                lastval += value
            else:

                # not the same token as last iteration, but we
                # have some data in the buffer. wrap it with the
                # defined style and write it to the output file
                if lastval:
                    stylebegin, styleend = self.styles[lasttype]

                    outfile.write(stylebegin + lastval + styleend)
                # set lastval/lasttype to current values

                lastval = value
                lasttype = ttype

        # if something is left in the buffer, write it to the

        # output file, then close the opened <pre> tag
        if lastval:
            stylebegin, styleend = self.styles[lasttype]

            outfile.write(stylebegin + lastval + styleend)
        outfile.write('</pre>\n')

Generating Style Definitions

Some formatters like the LatexFormatter and the HtmlFormatter don't output inline markup but reference either macros or css classes. Because the definitions of those are not part of the output, the get_style_defs() method exists. It is passed one parameter (if it's used and how it's used is up to the formatter) and has to return a string or None.
